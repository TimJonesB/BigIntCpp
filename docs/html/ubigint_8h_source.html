77inline UBigInt::UBigInt(std::string s) {
78 std::unordered_map<char,int> s2num{{'0',0}, {'1',1}, {'2',2},
79 {'3',3}, {'4',4}, {'5',5},
80 {'6',6}, {'7',7}, {'8',8},{'9',9}};
81 int i = 0;
82 while (s[i] == ' ') {
83 i++;
84 }
85 while (s[i] == '0' || s[i] == '+' || s[i] == '-') {
86 i++;
87 }
88 while (i < s.size()) {
89 if (s2num.find(s[i]) == s2num.end()) {
90 throw std::runtime_error("Invalid character in string");
91 }
92 num.push_back(s2num[s[i]]);
93 i++;
94 }
95 if (num.empty()) {
96 num = {0};
97 }

107inline UBigInt::UBigInt(T rhs) {
108 std::stack<int> st;
109 if (rhs == 0) {
110 num = {0};
111 }
112 if (rhs < 0) {
113 rhs = -1 * rhs;
114 }
115 while (rhs > 0) {
116 int pop = rhs % 10;
117 st.push(pop);
118 rhs = rhs/10;
119 }
120 while (!st.empty()) {
121 num.push_back(st.top());
122 st.pop();
123 }
124}

248inline UBigInt& UBigInt::operator+=(const UBigInt &rhs) {
249 int carry = 0;
250 for (int i = 0; i < std::max(num.size(), rhs.num.size()); i++) {
251 int column_sum = carry;
252 if (i < rhs.num.size()) {
253 column_sum += rhs.num[rhs.num.size()-i-1];
254 }
255 if (i < num.size()) {
256 column_sum += num[num.size()-i-1];
257 }
258 else {
259 num.push_front(0);
260 }
261 num[num.size()-i-1] = column_sum > 9 ? (column_sum - 10) : column_sum;
262 carry = (column_sum > 9);
263 }
264 if (carry) {
265 num.push_front(carry);
266 }
267 return *this;

277 int borrow = 0;

424inline UBigInt UBigInt::long_multiplication(const UBigInt &lhs, const UBigInt &rhs) {
425 if (lhs == 0 || rhs == 0) {
426 return 0;
427 }
428 const std::deque<int> &top = num.size() >= rhs.num.size() ? num : rhs.num;
429 const std::deque<int> &bottom = num.size() < rhs.num.size() ? num : rhs.num;
430 std::deque<int> product;
431 int carry_mult;
432 int carry_add;
433 int offset;
434 for (int j = 0; j < bottom.size(); j++) {
435 carry_mult = 0;
436 carry_add = 0;
437 for(int i = 0; i < top.size(); i++) {
438 offset = i + j;
439 int prod = (top[top.size()-i-1] * bottom[bottom.size()-j-1]) + carry_mult;
440 int num_mult = prod % 10;
441 carry_mult = prod / 10;
442 if(product.size() < offset+1) {
443 product.push_front(0);
444 }
445 int sum = product[product.size()-offset-1] + num_mult + carry_add;
446 int num_add = (sum > 9) ? (sum - 10) : sum;
447 carry_add = (sum > 9);
448 product[product.size()-offset-1] = num_add;
449 }
450 if (carry_mult || carry_add) {
451 product.push_front(carry_mult + carry_add);
452 }
453 }
454 return UBigInt(product.begin(), product.end());
455}

479inline UBigInt UBigInt::long_division(const UBigInt &rhs) {
480 UBigInt temp{0};
481 UBigInt sol;
482 if (rhs > *this) {
483 return UBigInt{0};
484 }
485 else if (rhs == *this) {
486 return UBigInt{1};
487 }
488 else {
489 for (int i = 0; i < num.size();) {
490 temp = temp * 10 + num[i++];
491 UBigInt quo = temp.divide_primative(rhs);
492 for (const auto it : quo.num) {
493 if (!(sol.num.empty() && quo == 0)) {
494 sol.num.push_back(it);
495 }
496 }
497 if (quo*rhs <= temp) {
498 temp = temp - (quo * rhs);
499 }
500 }
501 if (sol.num.empty()) {
502 return UBigInt{0};
503 }
504 else {
505 return sol;
506 }
507 }
