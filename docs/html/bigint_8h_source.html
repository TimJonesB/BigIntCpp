425inline BigInt BigInt::karatsuba(BigInt lhs, BigInt rhs) {
426 if (lhs.get_length() < karatsuba_thres || rhs.get_length() < karatsuba_thres) {
427 return lhs.magnitude *= rhs.magnitude;
428 }
429 int m = std::max(lhs.get_length(), rhs.get_length());
430 int m2 = std::ceil(m / 2);
431
432 BigInt high1, low1, high2, low2;
433 if (lhs.magnitude.get_length() < m2) {
434 high1 = 0;
435 low1 = lhs;
436 }
437 else {
438 high1 = lhs.magnitude.get_slice(0, lhs.magnitude.get_length() - m2);
439 low1 = lhs.magnitude.get_slice(lhs.magnitude.get_length() - m2, lhs.magnitude.get_length());
440 }
441 if (rhs.magnitude.get_length() < m2) {
442 high2 = 0;
443 low2 = rhs;
444 }
445 else {
446 high2 = rhs.magnitude.get_slice(0, rhs.magnitude.get_length() - m2);
447 low2 = rhs.magnitude.get_slice(rhs.magnitude.get_length() - m2, rhs.magnitude.get_length());
448 }
449
450 BigInt z0 = karatsuba(low1, low2);
451 BigInt z1 = karatsuba((low1 + high1), (low2 + high2));
452 BigInt z2 = karatsuba(high1, high2);
453
454 return ((z2.shift10(m2*2)) + ((z1-z2-z0).shift10(m2)) + z0);
455}
