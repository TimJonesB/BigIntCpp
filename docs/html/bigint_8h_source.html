463inline BigInt BigInt::karatsuba(BigInt lhs, BigInt rhs) {
464 if (lhs.get_length() < karatsuba_thres || rhs.get_length() < karatsuba_thres) {
465 return lhs.magnitude *= rhs.magnitude;
466 }
467 int m = std::max(lhs.get_length(), rhs.get_length());
468 int m2 = std::ceil(m / 2);
469
470 BigInt high1, low1, high2, low2;
471 if (lhs.magnitude.get_length() < m2) {
472 high1 = 0;
473 low1 = lhs;
474 }
475 else {
476 high1 = lhs.magnitude.get_slice(0, lhs.magnitude.get_length() - m2);
477 low1 = lhs.magnitude.get_slice(lhs.magnitude.get_length() - m2, lhs.magnitude.get_length());
478 }
479 if (rhs.magnitude.get_length() < m2) {
480 high2 = 0;
481 low2 = rhs;
482 }
483 else {
484 high2 = rhs.magnitude.get_slice(0, rhs.magnitude.get_length() - m2);
485 low2 = rhs.magnitude.get_slice(rhs.magnitude.get_length() - m2, rhs.magnitude.get_length());
486 }
487
488 BigInt z0 = karatsuba(low1, low2);
489 BigInt z1 = karatsuba((low1 + high1), (low2 + high2));
490 BigInt z2 = karatsuba(high1, high2);
491
492 return ((z2.shift10(m2*2)) + ((z1-z2-z0).shift10(m2)) + z0);
493}
